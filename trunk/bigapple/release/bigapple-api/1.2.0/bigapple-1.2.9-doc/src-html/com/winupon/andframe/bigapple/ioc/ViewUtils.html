059        private static void injectObject(Object handler, ViewFinder finder) {
060            Field[] fields = handler.getClass().getDeclaredFields();
061            if (fields != null && fields.length > 0) {
062                for (Field field : fields) {
063                    InjectView injectView = field.getAnnotation(InjectView.class);
064                    if (null != injectView) {
065                        try {
066                            View view = finder.findViewById(injectView.value());
067                            if (null != view) {
068                                field.setAccessible(true);
069                                field.set(handler, view);
070                            }
071                        }
