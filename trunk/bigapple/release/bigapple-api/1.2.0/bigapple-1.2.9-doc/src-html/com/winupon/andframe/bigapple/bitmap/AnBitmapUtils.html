105        public void display(ImageView imageView, String uri, BitmapDisplayConfig displayConfig) {
106            if (null == imageView) {
107                LogUtils.e("å›¾ç‰‡åŠ è½½ä¸å¤„ç†ï¼ŒåŸå› ï¼šå›¾ç‰‡æ˜¾ç¤ºæ§ä»¶imageViewä¸ºç©º");
108                return;
109            }
110    
111            if (null == displayConfig) {
112                displayConfig = defaultDisplayConfig;
113            }
114    
115            if (TextUtils.isEmpty(uri)) {
116                displayConfig.getImageLoadCallBack().loadFailed(imageView, displayConfig.getLoadFailedBitmap());
117                return;
118            }
119    
120            // å†…å­˜ç¼“å­˜ä¸­å–
121            Bitmap bitmap = globalConfig.getBitmapCache().getBitmapFromMemCache(uri, displayConfig);
122    
123            if (null != bitmap) {
124                // å†…å­˜ç¼“å­˜å‘½ä¸­
125                displayConfig.getImageLoadCallBack().loadCompleted(imageView, bitmap, displayConfig);
126            }
127            else if (!bitmapLoadTaskExist(imageView, uri)) {
128                // å¯åŠ¨ä»»åŠ¡ç±»ï¼šä»ç½‘ç»œä¸‹è½½æˆ–è€…ä»ç£ç›˜ä¸­è·å?
129                final BitmapLoadTask loadTask = new BitmapLoadTask(imageView, displayConfig);
130    
131                final AsyncBitmapDrawable asyncBitmapDrawable = new AsyncBitmapDrawable(application.getResources(),
132                        displayConfig.getLoadingBitmap(), loadTask);
133                imageView.setImageDrawable(asyncBitmapDrawable);// è®¾ç½®ä¸‹è½½ä»»åŠ¡èµ„æº
134    
135                loadTask.executeOnExecutor(globalConfig.getBitmapLoadExecutor(), uri);
136            }
137        }

338        private static BitmapLoadTask getBitmapTaskFromImageView(ImageView imageView) {
339            if (imageView != null) {
340                final Drawable drawable = imageView.getDrawable();
341                if (drawable instanceof AsyncBitmapDrawable) {
342                    final AsyncBitmapDrawable asyncBitmapDrawable = (AsyncBitmapDrawable) drawable;
343                    return asyncBitmapDrawable.getBitmapWorkerTask();
344                }
345            }
346            return null;
347        }
348    
349        private static boolean bitmapLoadTaskExist(ImageView imageView, String uri) {
350            final BitmapLoadTask oldLoadTask = getBitmapTaskFromImageView(imageView);
351    
352            if (oldLoadTask != null) {
353                final String oldUri = oldLoadTask.uri;
354                if (TextUtils.isEmpty(oldUri) || !oldUri.equals(uri)) {
355                    oldLoadTask.cancel(true);
356                }
357                else {
358                    // åŒä¸€ä¸ªçº¿ç¨‹å·²ç»åœ¨æ‰§è¡Œ
359                    return true;
360                }
361            }
362            return false;
363        }

385        /**
386         * å›¾ç‰‡åŠ è½½ä»»åŠ¡
387         * 
388         * @author xuan
389         * @version $Revision: 1.0 $, $Date: 2013-9-17 ä¸‹åˆ2:45:22 $
390         */
391        private class BitmapLoadTask extends CompatibleAsyncTask<Object, Void, Bitmap> {
392            private String uri;
393            private final WeakReference<ImageView> targetImageViewReference;
394            private final BitmapDisplayConfig displayConfig;
395    
396            public BitmapLoadTask(ImageView imageView, BitmapDisplayConfig config) {
397                targetImageViewReference = new WeakReference<ImageView>(imageView);
398                displayConfig = config;
399            }
400    
401            @Override
402            protected Bitmap doInBackground(Object... params) {
403                if (params != null && params.length > 0) {
404                    uri = (String) params[0];
405                }
406                else {
407                    return null;
408                }
409    
410                Bitmap bitmap = null;
411                synchronized (pauseTaskLock) {
412                    while (pauseTask && !this.isCancelled()) {
413                        try {
414                            pauseTaskLock.wait();
415                        }
416                        catch (InterruptedException e) {
417                        }
418                    }
419                }
420    
421                /* å…ˆè¯•å›¾ä»ç£ç›˜ç¼“å­˜ä¸­è¯»å–å›¾ç‰?*/
422                if (!pauseTask && !this.isCancelled() && this.getTargetImageView() != null) {
423                    bitmap = globalConfig.getBitmapCache().getBitmapFromDiskCache(uri, displayConfig);
424                }
425    
426                /* å½“ç£ç›˜ç¼“å­˜ä¸­çš„å›¾ç‰‡ä¸å­˜åœ¨äº†ï¼Œå°±å»ç½‘ç»œä¸ŠåŠ è½½éœ€è¦çš„å›¾ç‰‡ */
427                if (bitmap == null && !pauseTask && !this.isCancelled() && this.getTargetImageView() != null) {
428                    bitmap = globalConfig.getBitmapCache().downloadBitmap(uri, displayConfig);
429                }
430    
431                return bitmap;
432            }
433    
434            // è·å–å›¾ç‰‡ä»»åŠ¡å®Œæˆ
435            @Override
436            protected void onPostExecute(Bitmap bitmap) {
437                if (isCancelled() || pauseTask) {
438                    bitmap = null;
439                }
440    
441                final ImageView imageView = this.getTargetImageView();
442                if (imageView != null) {
443                    if (bitmap != null) {
444                        // æ˜¾ç¤ºå›¾ç‰‡
445                        displayConfig.getImageLoadCallBack().loadCompleted(imageView, bitmap, displayConfig);
446                    }
447                    else {
448                        // æ˜¾ç¤ºè·å–é”™è¯¯å›¾ç‰‡
449                        displayConfig.getImageLoadCallBack().loadFailed(imageView, displayConfig.getLoadFailedBitmap());
450                    }
451                }
452            }
453    
454            @Override
455            protected void onCancelled(Bitmap bitmap) {
456                super.onCancelled(bitmap);
457                synchronized (pauseTaskLock) {
458                    pauseTaskLock.notifyAll();
459                }
460            }
461    
462            /**
463             * è·å–çº¿ç¨‹åŒ¹é…çš„imageView,é˜²æ­¢å‡ºç°é—ªåŠ¨çš„ç°è±?
464             * 
465             * @return
466             */
467            private ImageView getTargetImageView() {
468                final ImageView imageView = targetImageViewReference.get();
469                final BitmapLoadTask bitmapWorkerTask = getBitmapTaskFromImageView(imageView);
470    
471                if (this == bitmapWorkerTask) {
472                    return imageView;
473                }
474    
475                return null;
476            }
477        }
