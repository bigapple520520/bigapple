051        /**
052         * å¼?§‹å½•åˆ¶
053         * 
054         * @param voiceFileName
055         * @param onRecordStartedListener
056         */
057        public void startRecord(String voiceFileName, final OnRecordStartedListener onRecordStartedListener) {
058            fileName = voiceFileName;
059    
060            singleThreadPool.submit(new Runnable() {
061                @Override
062                public void run() {
063                    if (isStarted) {
064                        LogUtils.d("startRecord isStarted");
065                        return;
066                    }
067    
068                    try {
069                        isStarted = true;
070                        LogUtils.d("startRecord1");
071                        getMediaRecorder();
072    
073                        mediaRecorder.setOnInfoListener(new OnInfoListener() {
074                            @Override
075                            public void onInfo(MediaRecorder mr, int what, int extra) {
076                                LogUtils.d("info:" + what + "," + extra);
077                            }
078                        });
079    
080                        mediaRecorder.setOnErrorListener(new OnErrorListener() {
081                            @Override
082                            public void onError(MediaRecorder mr, int what, int extra) {
083                                LogUtils.d("error:" + what + "," + extra);
084                            }
085                        });
086    
087                        mediaRecorder.setAudioSource(MediaRecorder.AudioSource.MIC);
088                        mediaRecorder.setOutputFormat(MediaRecorder.OutputFormat.DEFAULT);
089    
090                        // PreparingçŠ¶æ?
091                        mediaRecorder.setAudioEncoder(MediaRecorder.AudioEncoder.DEFAULT);
092                        mediaRecorder.setOutputFile(mediaConfig.getVoicePath() + File.separator + fileName + "."
093                                + mediaConfig.getVoiceExt());
094    
095                        try {
096                            mediaRecorder.prepare();
097                        }
098                        catch (Exception e) {
099                            LogUtils.e("", e);
100                        }
101    
102                        // PreparedçŠ¶æ?
103                        mediaRecorder.start(); // å¼?§‹å½•éŸ³
104    
105                        // é€šçŸ¥å½•éŸ³å¼?§‹
106                        handler.post(new Runnable() {
107                            @Override
108                            public void run() {
109                                onRecordStartedListener.onRecordStarted();
110                            }
111                        });
112    
113                        lastStartTimeMillis = System.currentTimeMillis();
114                        LogUtils.d("startRecord2");
115                    }
116                    catch (Exception e) {
117                        LogUtils.e("", e);
118                    }
119                }
120            });
121        }

124         * ç»“æŸå½•åˆ¶
125         * 
126         * @param onRecordStopedListener
127         */
128        public void stopRecord(final OnRecordStopedListener onRecordStopedListener) {
129            singleThreadPool.submit(new Runnable() {
130                @Override
131                public void run() {
132                    if (!isStarted) {
133                        LogUtils.d("stopRecord !isStarted");
134                        return;
135                    }
136    
137                    try {
138                        LogUtils.d("stopRecord1");
139                        long x = System.currentTimeMillis() - lastStartTimeMillis;
140                        boolean success = true;
141    
142                        // å½•éŸ³æ—¶é—´å¦‚æœæ²¡è¶…è¿?Sè¿›è¡ŒonTooShorté€šçŸ¥
143                        if (x < 1000) {
144                            success = false;
145    
146                            handler.post(new Runnable() {
147                                @Override
148                                public void run() {
149                                    onRecordStopedListener.onTooShort();
150                                }
151                            });
152    
153                            try {
154                                Thread.sleep(1000);
155                            }
156                            catch (InterruptedException e) {
157                                LogUtils.e("", e);
158                            }
159                        }
160    
161                        getMediaRecorder();
162                        mediaRecorder.stop();
163                        mediaRecorder.reset(); // é‡ç½®mediaRecorderå¯¹è±¡ï¼Œä½¿å…¶ä¸ºç©ºé—²çŠ¶æ?
164                        mediaRecorder.release();// é‡Šæ”¾mediaRecorderå¯¹è±¡
165    
166                        // åˆ é™¤å½•éŸ³æ—¶é—´å¤ªçŸ­çš„æ–‡ä»?
167                        if (!success) {
168                            deleteFolderFile(
169                                    mediaConfig.getVoicePath() + File.separator + fileName + "."
170                                            + mediaConfig.getVoiceExt(), true);
171                        }
172    
173                        final boolean temp = success;
174    
175                        // é€šçŸ¥å½•éŸ³ç»“æŸ
176                        handler.post(new Runnable() {
177                            @Override
178                            public void run() {
179                                onRecordStopedListener.onRecordStoped(temp, fileName);
180                            }
181                        });
182    
183                        LogUtils.d("stopRecord2");
184                    }
185                    catch (Exception e) {
186                        LogUtils.e("", e);
187                    }
188                    finally {
189                        isStarted = false;
190                    }
191                }
192            });
193        }

227        private static void deleteFolderFile(String filePath, boolean deleteThisPath) throws IOException {
228            if (!TextUtils.isEmpty(filePath)) {
229                File file = new File(filePath);
230    
231                if (file.isDirectory()) {// å¤„ç†ç›®å½•
232                    File files[] = file.listFiles();
233                    for (int i = 0; i < files.length; i++) {
234                        deleteFolderFile(files[i].getAbsolutePath(), true);
235                    }
236                }
237    
238                if (deleteThisPath) {
239                    if (!file.isDirectory()) {// å¦‚æœæ˜¯æ–‡ä»¶ï¼Œåˆ é™¤
240                        file.delete();
241                    }
242                    else {// ç›®å½•
243                        if (file.listFiles().length == 0) {// ç›®å½•ä¸‹æ²¡æœ‰æ–‡ä»¶æˆ–è€…ç›®å½•ï¼Œåˆ é™¤
244                            file.delete();

247                }
