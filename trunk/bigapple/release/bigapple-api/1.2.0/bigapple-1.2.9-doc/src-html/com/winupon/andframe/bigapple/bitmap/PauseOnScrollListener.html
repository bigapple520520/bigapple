012    public class PauseOnScrollListener implements OnScrollListener {
013        private final AnBitmapUtils anBitmapUtils;
014        /**
015         * æ ‡è¯†æ»šåŠ¨æ—¶æ˜¯å¦åœæ­¢ä¸‹è½?     */
016        private final boolean pauseOnScroll;
017    
018        /**
019         * æ ‡è¯†æ‰‹æŒ‡åšæŠ›çš„åŠ¨ä½œæ—¶ï¼Œæ˜¯å¦åœæ­¢ä¸‹è½?     */
020        private final boolean pauseOnFling;
021    
022        /**
023         * å¤–éƒ¨è‡ªå®šä¹‰çš„OnScrollListenerç›‘å¬å®ç°
024         */
025        private final OnScrollListener externalListener;
026    
027        public PauseOnScrollListener(AnBitmapUtils anBitmapUtils, boolean pauseOnScroll, boolean pauseOnFling) {
028            this(anBitmapUtils, pauseOnScroll, pauseOnFling, null);
029        }
030    
031        public PauseOnScrollListener(AnBitmapUtils anBitmapUtils, boolean pauseOnScroll, boolean pauseOnFling,
032                OnScrollListener customListener) {
033            this.anBitmapUtils = anBitmapUtils;
034            this.pauseOnScroll = pauseOnScroll;
035            this.pauseOnFling = pauseOnFling;
036            externalListener = customListener;
037        }
038    
039        @Override
040        public void onScrollStateChanged(AbsListView view, int scrollState) {
041            switch (scrollState) {
042            case OnScrollListener.SCROLL_STATE_IDLE:
043                anBitmapUtils.resumeTasks();
044                break;
045            case OnScrollListener.SCROLL_STATE_TOUCH_SCROLL:
046                if (pauseOnScroll) {
047                    anBitmapUtils.pauseTasks();
048                }
049                break;
050            case OnScrollListener.SCROLL_STATE_FLING:
051                if (pauseOnFling) {
052                    anBitmapUtils.pauseTasks();
053                }
054                break;
055            }
056    
057            if (externalListener != null) {
058                externalListener.onScrollStateChanged(view, scrollState);
059            }
060        }
061    
062        @Override
063        public void onScroll(AbsListView view, int firstVisibleItem, int visibleItemCount, int totalItemCount) {
064            if (externalListener != null) {
065                externalListener.onScroll(view, firstVisibleItem, visibleItemCount, totalItemCount);
066            }
067        }
068    
069    }
