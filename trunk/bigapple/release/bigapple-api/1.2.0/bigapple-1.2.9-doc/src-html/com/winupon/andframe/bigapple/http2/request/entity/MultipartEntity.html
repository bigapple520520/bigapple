022    public class MultipartEntity implements RequestEntiy {
023            private final static char[] MULTIPART_CHARS = "-_1234567890abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ"
024                            .toCharArray();
025    
026            private String boundary = null;// ËæπÁïå
027            ByteArrayOutputStream out = new ByteArrayOutputStream();
028    
029            boolean isSetLast = false;
030            boolean isSetFirst = false;
031    
032            private final RequestParams requestParams;
033    
034            public MultipartEntity(RequestParams requestParams) {
035                    initBoundary();
036                    this.requestParams = requestParams;
037            }
038    
039            // ÂàùÂßãÂåñ‰∏Ä‰∏™ÈöèÊú∫ÂàÜÈöîÁ¨¶
040            private void initBoundary() {
041                    final StringBuilder sb = new StringBuilder();
042                    final Random rand = new Random();
043                    for (int i = 0; i < 30; i++) {
044                            sb.append(MULTIPART_CHARS[rand.nextInt(MULTIPART_CHARS.length)]);
045                    }
046                    this.boundary = sb.toString();
047            }
048    
049            /**
050             * ËÆæÁΩÆÁ¨¨‰∏Ä‰∏™ÂàÜÁï?
051             */
052            public void writeFirstBoundaryIfNeeds() {
053                    if (isSetFirst) {
054                            return;
055                    }
056    
057                    try {
058                            out.write(("--" + boundary + "\r\n").getBytes());
059                    } catch (IOException e) {
060                            LogUtils.e("", e);
061                    }
062    
063                    isSetFirst = true;
064            }
065    
066            /**
067             * ËÆæÁΩÆÊú?êé‰∏?∏™ÂàÜÁïå
068             */
069            public void writeLastBoundaryIfNeeds() {
070                    if (isSetLast) {
071                            return;
072                    }
073    
074                    try {
075                            out.write(("\r\n--" + boundary + "--\r\n").getBytes());
076                    } catch (final IOException e) {
077                            LogUtils.e("", e);
078                    }
079    
080                    isSetLast = true;
081            }
082    
083            /**
084             * Ê∑ªÂä†ÊôÆÈ?ÂèÇÊï∞
085             * 
086             * @param key
087             * @param value
088             */
089            public void addPart(final String key, final String value) {
090                    writeFirstBoundaryIfNeeds();
091                    try {
092                            out.write(("Content-Disposition: form-data; name=\"" + key + "\"\r\n\r\n")
093                                            .getBytes());
094                            out.write(value.getBytes());
095                            out.write(("\r\n--" + boundary + "\r\n").getBytes());
096                    } catch (IOException e) {
097                            LogUtils.e("", e);
098                    }
099            }
100    
101            /**
102             * Ê∑ªÂä†Êñá‰ª∂
103             * 
104             * @param key
105             * @param fileName
106             * @param fin
107             * @param isLast
108             */
109            public void addPart(final String key, final String fileName,
110                            final InputStream fin, final boolean isLast) {
111                    addPart(key, fileName, fin, "application/octet-stream", isLast);
112            }
113    
114            /**
115             * Ê∑ªÂä†Êñá‰ª∂
116             * 
117             * @param key
118             * @param fileName
119             * @param fin
120             * @param type
121             * @param isLast
122             */
123            public void addPart(final String key, final String fileName,
124                            final InputStream fin, String type, final boolean isLast) {
125                    writeFirstBoundaryIfNeeds();
126                    try {
127                            type = "Content-Type: " + type + "\r\n";
128                            out.write(("Content-Disposition: form-data; name=\"" + key
129                                            + "\"; filename=\"" + fileName + "\"\r\n").getBytes());
130                            out.write(type.getBytes());
131                            out.write("Content-Transfer-Encoding: binary\r\n\r\n".getBytes());
132    
133                            final byte[] tmp = new byte[4 * 1024];
134                            int l = 0;
135                            while ((l = fin.read(tmp)) != -1) {
136                                    out.write(tmp, 0, l);
137                            }
138                            if (!isLast) {
139                                    out.write(("\r\n--" + boundary + "\r\n").getBytes());
140                            }
141                            out.flush();
142                    } catch (final IOException e) {
143                            LogUtils.e("", e);
144                    } finally {
145                            IOUtils.closeQuietly(fin);
146                    }
147            }
148    
149            @Override
150            public void writeTo(final OutputStream outstream) throws IOException {
151                    Map<String, String> paramMap = null;
152                    Map<String, FileWraper> fileParamMap = null;
153    
154                    if (null != requestParams) {
155                            paramMap = requestParams.getParamMap();
156                            fileParamMap = requestParams.getFileParamMap();
157                    }
158    
159                    // Ê∑ªÂä†ÊôÆÈ?ÂèÇÊï∞
160                    if (null != paramMap && !paramMap.isEmpty()) {
161                            for (ConcurrentHashMap.Entry<String, String> entry : paramMap
162                                            .entrySet()) {
163                                    addPart(entry.getKey(), entry.getValue());
164                            }
165                    }
166    
167                    // Ê∑ªÂä†Êñá‰ª∂ÂèÇÊï∞
168                    if (null != fileParamMap && !fileParamMap.isEmpty()) {
169                            int currentIndex = 0;
170                            int lastIndex = fileParamMap.entrySet().size() - 1;
171                            for (ConcurrentHashMap.Entry<String, FileWraper> entry : fileParamMap
172                                            .entrySet()) {
173                                    FileWraper file = entry.getValue();
174                                    if (file.inputStream != null) {
175                                            boolean isLast = (currentIndex == lastIndex);
176                                            if (file.contentType != null) {
177                                                    addPart(entry.getKey(), file.getFileName(),
178                                                                    file.inputStream, file.contentType, isLast);
179                                            } else {
180                                                    addPart(entry.getKey(), file.getFileName(),
181                                                                    file.inputStream, isLast);
182                                            }
183                                    }
184                                    currentIndex++;
185                            }
186                    }
187    
188                    writeLastBoundaryIfNeeds();
189                    outstream.write(out.toByteArray());
190            }
191    
192            public String getBoundary() {
193                    return boundary;
194            }
195    
196            public void setBoundary(String boundary) {
197                    this.boundary = boundary;
198            }
