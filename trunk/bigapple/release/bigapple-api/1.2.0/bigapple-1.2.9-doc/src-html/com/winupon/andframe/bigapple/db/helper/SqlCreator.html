084        public void addExpression(String operator, String expression, String arg, boolean precondition) {
085            if (precondition) {
086                if (isFirst) {
087                    if (hasWhere) {
088                        if (!sql.toString().toLowerCase().endsWith("where")) {

090                        }
091                    }

094                    }
095                    isFirst = false;
096                }
097                else {

099                }

102    
103                if (arg != null) {
104                    args.add(arg);
105                }
106            }
