067         * @param separator
068         * @param startIndex
069         * @param endIndex
070         * @return
071         */
072        public static String join(Object[] array, String separator, int startIndex, int endIndex) {
073            if (array == null) {
074                return null;
075            }
076    
077            int noOfItems = endIndex - startIndex;
078            if (noOfItems <= 0) {
079                return "";
080            }
081    
082            StringBuilder buf = new StringBuilder(noOfItems * 16);
083    
084            for (int i = startIndex; i < endIndex; i++) {
085                if (i > startIndex) {
086                    buf.append(separator);
087                }
088                if (array[i] != null) {
089                    buf.append(array[i]);
090                }
091            }
092            return buf.toString();
093        }

102        public static String join(Iterator<?> iterator, String separator) {
103            if (iterator == null) {
104                return null;
105            }
106            if (!iterator.hasNext()) {
107                return "";
108            }
109            Object first = iterator.next();
110            if (!iterator.hasNext()) {
111                return first == null ? "" : first.toString();
112            }
113    
114            // å«æœ‰ä¸¤ä¸ªä»¥ä¸Šçš„å…ƒç´?
115            StringBuilder buf = new StringBuilder(256);// javaé»˜è®¤æ˜?6ä¸ªï¼Œå¯èƒ½å¤ªå°äº?
116            if (first != null) {
117                buf.append(first);
118            }
119    
120            while (iterator.hasNext()) {
121                if (separator != null) {
122                    buf.append(separator);
123                }
124                Object obj = iterator.next();
125                if (obj != null) {
126                    buf.append(obj);
127                }
128            }
129            return buf.toString();
130        }

156        public static String[] splitWorker(String str, String separatorChars, int max, boolean preserveAllTokens) {
157            if (str == null) {
158                return null;
159            }
160    
161            int len = str.length();
162            if (len == 0) {
163                return new String[0];
164            }
165    
166            List<String> list = new ArrayList<String>();
167            int sizePlus1 = 1;
168            int i = 0, start = 0;
169            boolean match = false;// trueè¡¨ç¤ºæœ‰è¦åˆ†éš”çš„å†…å®?
170            boolean lastMatch = false;// trueè¡¨ç¤ºåŒ¹é…åˆ°åˆ†éš”ç¬¦
171            if (separatorChars == null) {
172                // nullä½œä¸ºåˆ†éš”ç¬¦å°±è¡¨ç¤ºç”¨ç©ºæ ¼ä½œä¸ºåˆ†éš?
173                while (i < len) {
174                    if (Character.isWhitespace(str.charAt(i))) {
175                        if (match || preserveAllTokens) {
176                            lastMatch = true;
177                            if (sizePlus1++ == max) {
178                                i = len;
179                                lastMatch = false;
180                            }
181                            list.add(str.substring(start, i));
182                            match = false;
183                        }
184                        start = ++i;
185                        continue;
186                    }
187                    lastMatch = false;
188                    match = true;
189                    i++;
190                }
191            }
192            else if (separatorChars.length() == 1) {
193                // åˆ†éš”ç¬¦ä¸ºä¸?¸ªå­—ç¬¦çš„æƒ…å†µä¸‹
194                char sep = separatorChars.charAt(0);
195                while (i < len) {
196                    if (str.charAt(i) == sep) {
197                        if (match || preserveAllTokens) {
198                            lastMatch = true;
199                            if (sizePlus1++ == max) {
200                                i = len;
201                                lastMatch = false;
202                            }
203                            list.add(str.substring(start, i));
204                            match = false;
205                        }
206                        start = ++i;
207                        continue;
208                    }
209                    lastMatch = false;
210                    match = true;
211                    i++;
212                }
213            }
214            else {
215                // æ ‡å‡†æƒ…å†µ
216                while (i < len) {
217                    if (separatorChars.indexOf(str.charAt(i)) >= 0) {
218                        if (match || preserveAllTokens) {
219                            lastMatch = true;
220                            if (sizePlus1++ == max) {
221                                i = len;
222                                lastMatch = false;
223                            }
224                            list.add(str.substring(start, i));
225                            match = false;
226                        }
227                        start = ++i;
228                        continue;
229                    }
230                    lastMatch = false;
231                    match = true;
232                    i++;
233                }
234            }
235    
236            // å¦‚æœæœ‰åŒ¹é…å†…å®?|| åŒ¹é…åˆ°åˆ†éš”ç¬¦ä¸”é¦–å°¾åˆ†éš”ç¬¦è¦ç©ºæ ?
237            if (match || (preserveAllTokens && lastMatch)) {
238                list.add(str.substring(start, i));
239            }
240    
241            return list.toArray(new String[list.size()]);
242        }

300         * æˆªå–å›ºå®šé•¿åº¦çš„å­—ç¬¦ä¸²ï¼Œè¶…é•¿éƒ¨åˆ†ç”¨suffixä»£æ›¿ï¼Œæœ€ç»ˆå­—ç¬¦ä¸²çœŸå®é•¿åº¦ä¸ä¼šè¶…è¿‡maxLength.
301         * 
302         * @param str
303         * @param maxLength
304         * @param suffix
305         * @return
306         */
307        public static String cutOut(String str, int maxLength, String suffix) {
308            if (Validators.isEmpty(str)) {
309                return str;
310            }
311    
312            int byteIndex = 0;
313            int charIndex = 0;
314    
315            while (charIndex < str.length() && byteIndex <= maxLength) {
316                char c = str.charAt(charIndex);
317                if (c >= 256) {
318                    byteIndex += 2;
319                }
320                else {
321                    byteIndex++;
322                }
323                charIndex++;
324            }
325    
326            if (byteIndex <= maxLength) {
327                return str;
328            }
329    
330            StringBuilder sb = new StringBuilder();
331            sb.append(str.substring(0, charIndex));
332            sb.append(suffix);
333    
334            while (getRealLength(sb.toString()) > maxLength) {
335                sb.deleteCharAt(--charIndex);
336            }
337    
338            return sb.toString();
339        }
